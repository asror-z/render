<form id="form1" action="action.php?delay=1" method="post" class="panel panel-default">
    <div class="panel-heading">Form 1</div>
    <div class="panel-body">
        <button type="submit">Submit</button>

<script>
$(document).on('submitlast',   '#form1', function () { $(this).find('pre').append("submitlast\n"); });
$(document).on('submitbefore', '#form1', function (e) { e.preventDefault(); $(this).find('pre').append("submitbefore\n"); });
$(document).on('submitstart',  '#form1', function () { $(this).find('pre').append("submitstart\n"); });
$(document).on('submitend',    '#form1', function () { $(this).find('pre').append("submitend\n"); });
$(document).on('submit',       '#form1', function () { $(this).find('pre').append("submit\n"); });
