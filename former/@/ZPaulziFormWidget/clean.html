<script>
window.PaulZiForm = $.extend(true, window.PaulZiForm || {}, {
    buttonLoadingTemplate: function (data) {
        if (data.btn.id === 'btn-template') {
            return 'My template (text = ' + data.text + ')';
        } else if (data.btn.id === 'btn-notemplate') {
            return false;
        } else {
            return window.PaulZiForm.defaultTemplate(data);
        }
    }
});
</script>
