<form id="form-state" action="action.php?delay=1" method="post" class="panel panel-default">
    <div class="panel-heading">Form state <span class="state">(please wait)</span></div>
    <div class="panel-body">
        Your name: <input type="text" name="name" value="John"> <small>(try to submit with ENTER)</small><br>
        <button type="submit" data-loading-icon="fa fa-refresh fa-spin" data-loading-text=" Exporting...">Export</button>
        <button type="submit" data-loading-text="Saving..." id="btn-template">Save (custom template)</button>

    </div>

<script>
window.PaulZiForm = $.extend(true, window.PaulZiForm || {}, {
    buttonLoadingTemplate: function (data) {
        if (data.btn.id === 'btn-template') {
            return 'My template (text = ' + data.text + ')';
        } else if (data.btn.id === 'btn-notemplate') {
            return false;
        } else {
            return window.PaulZiForm.defaultTemplate(data);
        }
    }
});
</script>

    <div class="panel-heading">Form ajax</div>
    <div class="panel-body">
        Your name: <input type="text" name="name" value="John"><br>

        <input type="image" name="image" src="i/btn-image.png">
        <button type="submit" data-loading-text="..." formaction="action.php?action=error">Error</button>
    </div>
    <pre class="events"></pre>
    <div class="my-output"></div>
</form>
<script>
$(document).on('submitlast',       '#form-ajax', function () { $(this).find('.events').append("submitlast\n"); });
$(document).on('submitajax',       '#form-ajax', function () { $(this).find('.events').append("submitajax\n"); });
$(document).on('submitbefore',     '#form-ajax', function (e) { e.preventDefault(); $(this).find('.events').append("submitbefore\n"); });
$(document).on('submitstart',      '#form-ajax', function () { $(this).find('.events').append("submitstart\n"); });
$(document).on('submitdone',       '#form-ajax', function () { $(this).find('.events').append("submitdone\n"); });
$(document).on('uploadprogress',   '#form-ajax', function (e) { $(this).find('.events').append("uploadprogress loaded=" + e.loaded +" total=" + e.total + "\n"); });
$(document).on('uploadend',        '#form-ajax', function (e) { $(this).find('.events').append("uploadend loaded=" + e.loaded +" total=" + e.total + "\n"); });
$(document).on('downloadprogress', '#form-ajax', function (e) { $(this).find('.events').append("downloadprogress loaded=" + e.loaded +" total=" + e.total + "\n"); });
$(document).on('submitfail',       '#form-ajax', function () { $(this).find('.events').append("submitfail\n"); });
$(document).on('submitend',        '#form-ajax', function () { $(this).find('.events').append("submitend\n"); });
$(document).on('submit',           '#form-ajax', function () { $(this).find('.events').append("submit\n"); });
</script>
